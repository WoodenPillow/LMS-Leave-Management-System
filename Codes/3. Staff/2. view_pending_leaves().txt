void view_pending_leaves() {
    if (leave_count == 0) {
        printf(nNo leaves foundn);
        return;
    }
    printf(IDtEmployee IDtStart DatetEnd DatetReasonn);
    for (int i = 0; i  leave_count; i++) {
        if (leaves[i].status == PENDING) {
            printf(%dt%dtt%st%st%sn, leaves[i].id, leaves[i].employee_id, leaves[i].start_date, leaves[i].end_date, leaves[i].reason);
        }
    }
}
