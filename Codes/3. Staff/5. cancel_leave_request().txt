    int leave_id;
    printf(Enter the leave ID of the leave request to cancel );
    scanf(%d, &leave_id);
    
    bool found = false;
    int leave_index = 0;
    for (int i = 0; i  leave_count; i++) {
        if (leaves[i].id == leave_id) {
            found = true;
            leave_index = i;
            break;
        }
    }
    if (!found) {
        printf(Error Leave request not foundn);
        return;
    }
    
    for (int i = leave_index; i  leave_count - 1; i++) {
        leaves[i] = leaves[i + 1];
    }

}
