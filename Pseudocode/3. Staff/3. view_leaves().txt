function view_leaves():
    if leave_count equals 0:
        print "No leaves found"
        return
    end if
    print "ID\tEmployee ID\tStart Date\tEnd Date\tReason\tStatus"
    for i from 0 to leave_count-1:
        let status be a character
        switch leaves[i].status:
            case PENDING:
                status = 'PENDING'
                break
            case APPROVED:
                status = 'APPROVED'
                break
            case REJECTED:
                status = 'REJECTED'
                break
        end switch
        print leaves[i].id, "\t", leaves[i].employee_id, "\t\t", leaves[i].start_date, "\t", leaves[i].end_date, "\t", leaves[i].reason, "\t", status
    end for
end function
